TABLES:
projects(
  id uuid pk default gen_random_uuid(),
  name text not null,
  created_at timestamptz not null default now()
);

pages(
  id uuid pk default gen_random_uuid(),
  project_id uuid not null references projects(id) on delete cascade,
  title text not null,
  status text not null default 'active',
  current_version int not null default 0,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

page_versions(
  id uuid pk default gen_random_uuid(),
  page_id uuid not null references pages(id) on delete cascade,
  version int not null,
  created_at timestamptz not null default now(),
  created_by text,
  snapshot bytea, -- optional compressed Yjs state (ydoc state vector snapshot)
  unique(page_id, version)
);

yjs_updates(
  id bigserial pk,
  page_id uuid not null references pages(id) on delete cascade,
  version int not null,
  created_at timestamptz not null default now(),
  update bytea not null
);
index yjs_updates_page_time on yjs_updates(page_id, created_at);

blocks(
  id uuid pk default gen_random_uuid(),
  page_id uuid not null references pages(id) on delete cascade,
  version int not null,
  ordinal int not null, -- order within page
  block_type text not null, -- heading, paragraph, list_item, code, table_row, etc.
  headings_path text, -- e.g. "H1 Title > H2 Section"
  canonical_json jsonb not null, -- structure+formatting
  plain_text text not null, -- formatting stripped
  render_html text, -- optional minimal
  created_at timestamptz not null default now(),
  unique(page_id, version, ordinal)
);
